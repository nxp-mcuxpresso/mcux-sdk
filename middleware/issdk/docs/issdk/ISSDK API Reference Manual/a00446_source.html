93 /* Handler for Device Info and Streaming Control Commands. */
94 bool process_host_command(
95  uint8_t tag, uint8_t *hostCommand, uint8_t *hostResponse, size_t *hostMsgSize, size_t respBufferSize)
96 {
97  bool success = false;
98 
99  /* If it is Host requesting Device Info, send Board Name and Shield Name. */
100  if (tag == HOST_PRO_INT_DEV_TAG)
101  { /* Byte 1 : Payload - Length of APPLICATION_NAME (b)
102  * Bytes=b : Payload Application Name
103  * Byte b+1 : Payload - Length of BOARD_NAME (s)
104  * Bytes=s : Payload Board Name
105  * Byte b+s+2 : Payload - Length of SHIELD_NAME (v)
106  * Bytes=v : Payload Shield Name */
107 
108  size_t appNameLen = strlen(embAppName);
109  size_t boardNameLen = strlen(boardString);
110  size_t shieldNameLen = strlen(shieldString);
111 
112  if (respBufferSize >= boardNameLen + shieldNameLen + appNameLen + 3)
113  { /* We have sufficient buffer. */
114  *hostMsgSize = 0;
115  }
116  else
117  {
118  return false;
119  }
120 
121  hostResponse[*hostMsgSize] = appNameLen;
122  *hostMsgSize += 1;
123 
124  memcpy(hostResponse + *hostMsgSize, embAppName, appNameLen);
125  *hostMsgSize += appNameLen;
126 
127  hostResponse[*hostMsgSize] = boardNameLen;
128  *hostMsgSize += 1;
129 
130  memcpy(hostResponse + *hostMsgSize, boardString, boardNameLen);
131  *hostMsgSize += boardNameLen;
132 
133  hostResponse[*hostMsgSize] = shieldNameLen;
134  *hostMsgSize += 1;
135 
136  memcpy(hostResponse + *hostMsgSize, shieldString, shieldNameLen);
137  *hostMsgSize += shieldNameLen;
138 
139  return true;
140  }
141 
142  /* If it is Host sending Streaming Commands, take necessary actions. */
143  if ((tag == (HOST_PRO_INT_CMD_TAG | HOST_PRO_CMD_W_CFG_TAG)) &&
144  (*hostMsgSize == HOST_MSG_CMD_ACT_OFFSET - HOST_MSG_LEN_LSB_OFFSET))
145  { /* Byte 1 : Payload - Operation Code (Start/Stop Operation Code)
146  * Byte 2 : Payload - Stream ID (Target Stream for carrying out operation) */
147  switch (hostCommand[0]) /* Execute desired operation (Start/Stop) on the requested Stream. */
148  {
149  case HOST_CMD_START:
150  if (hostCommand[1] == gStreamID && bMag3110Ready && bStreamingEnabled == false)
151  {
152  BOARD_SystickStart(&gSystick);
153  bStreamingEnabled = true;
154  success = true;
155  }
156  break;
157  case HOST_CMD_STOP:
158  if (hostCommand[1] == gStreamID && bMag3110Ready && bStreamingEnabled == true)
159  {
160  pGpioDriver->clr_pin(&GREEN_LED);
161  bStreamingEnabled = false;
162  success = true;
163  }
164  break;
165  default:
166  break;
167  }
168  *hostMsgSize = 0; /* Zero payload in response. */
169  }
170 
171  return success;
172 }

174 /*!
175  * @brief Main function
176  */
177 int main(void)
178 {
179  int32_t status;
180  uint8_t tempdata, dataReady, data[MAG3110_DATA_SIZE], streamingPacket[STREAMING_HEADER_LEN + MAG3110_STREAM_DATA_SIZE];
181 
182  mag3110_i2c_sensorhandle_t mag3110Driver;
183  mag3110_magdataUser_t rawData = {.timestamp = 0};
184 
185  ARM_DRIVER_I2C *pI2cDriver = &I2C_S_DRIVER;
186  ARM_DRIVER_USART *pUartDriver = &HOST_S_DRIVER;
187 
188  /*! Initialize the MCU hardware. */
189  BOARD_BootClockRUN();
190  BOARD_SystickEnable();
191 
192  /* Create the Short Application Name String for ADS. */
193  sprintf(embAppName, "%s:%s", APPLICATION_NAME, APPLICATION_VERSION);
194 
195  /* Run ADS. */
196  BOARD_RunADS(embAppName, boardString, shieldString, ADS_MAX_STRING_LENGTH);
197 
198  /* Create the Full Application Name String for Device Info Response. */
199  sprintf(embAppName, "%s:%s:%s", SHIELD_NAME, APPLICATION_NAME, APPLICATION_VERSION);
200 
201  /*! Initialize GREEN LED pin used by FRDM board */
202  pGpioDriver->pin_init(&GREEN_LED, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
203 
204  /*! Initialize the I2C driver. */
205  status = pI2cDriver->Initialize(I2C_S_SIGNAL_EVENT);
206  if (ARM_DRIVER_OK != status)
207  {
208  return -1;
209  }
210 
211  /*! Set the I2C Power mode. */
212  status = pI2cDriver->PowerControl(ARM_POWER_FULL);
213  if (ARM_DRIVER_OK != status)
214  {
215  return -1;
216  }
217 
218  /*! Set the I2C bus speed. */
219  status = pI2cDriver->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
220  if (ARM_DRIVER_OK != status)
221  {
222  return -1;
223  }
224 
225  /*! Initialize the UART driver. */
226  status = pUartDriver->Initialize(HOST_S_SIGNAL_EVENT);
227  if (ARM_DRIVER_OK != status)
228  {
229  return -1;
230  }
231 
232  /*! Set the UART Power mode. */
233  status = pUartDriver->PowerControl(ARM_POWER_FULL);
234  if (ARM_DRIVER_OK != status)
235  {
236  return -1;
237  }
238 
239  /*! Set UART Baud Rate. */
240  status = pUartDriver->Control(ARM_USART_MODE_ASYNCHRONOUS, BOARD_DEBUG_UART_BAUDRATE);
241  if (ARM_DRIVER_OK != status)
242  {
243  return -1;
244  }
245 
246  /*! Initialize the MAG3110 sensor driver. */
247  status = MAG3110_I2C_Initialize(&mag3110Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MAG3110_I2C_ADDR,
248  MAG3110_WHOAMI_VALUE);
249  if (SENSOR_ERROR_NONE == status)
250  {
251  /*! Set the task to be executed while waiting for I2C transactions to complete. */
252  MAG3110_I2C_SetIdleTask(&mag3110Driver, (registeridlefunction_t)SMC_SetPowerModeVlpr, SMC);
253 
254  /*! Configure the MAG3110 sensor driver. */
255  status = MAG3110_I2C_Configure(&mag3110Driver, cMag3110ConfigNormal);
256  if (SENSOR_ERROR_NONE == status)
257  {
258  bMag3110Ready = true;
259  }
260  }
261 
262  /*! Initialize streaming and assign a Stream ID. */
263  gStreamID =
264  Host_IO_Init(pUartDriver, (void *)mag3110Driver.pCommDrv, &mag3110Driver.deviceInfo, NULL, MAG3110_I2C_ADDR);
265  /* Confirm if a valid Stream ID has been allocated for this stream. */
266  if (0 == gStreamID)
267  {
268  bMag3110Ready = false;
269  }
270  else
271  {
272  /*! Populate streaming header. */
273  Host_IO_Add_ISO_Header(gStreamID, streamingPacket, MAG3110_STREAM_DATA_SIZE);
274  pGpioDriver->clr_pin(&GREEN_LED); /* Set LED to indicate application is ready. */
275  }
276 
277  for (;;) /* Forever loop */
278  { /* Call UART Non-Blocking Receive. */
279  Host_IO_Receive(process_host_command, HOST_FORMAT_HDLC);
280 
281  /* Process packets only if streaming has been enabled by Host. */
282  if (false == bStreamingEnabled)
283  {
284  SMC_SetPowerModeWait(SMC); /* Power save, wait if nothing to do. */
285  continue;
286  }
287  else
288  {
289  /*! Wait for data ready from the MAG3110. */
290  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110Status, &dataReady);
291  if (0 == (dataReady & MAG3110_DR_STATUS_ZYXDR_MASK))
292  { /* Loop, if new sample is not available. */
293  continue;
294  }
295  pGpioDriver->toggle_pin(&GREEN_LED);
296  }
297 
298  /*! Read raw sensor data from the MAG3110. */
299  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110OutputNormal, data);
300  if (ARM_DRIVER_OK != status)
301  { /* Loop, if sample read failed. */
302  continue;
303  }
304 
305  /* Update timestamp from Systick framework. */
306  rawData.timestamp += BOARD_SystickElapsedTime_us(&gSystick);
307 
308  /*! Convert the raw sensor data to signed 32-bit and 16-bit containers for display to the debug port. */
309  rawData.mag[0] = ((int16_t)data[0] << 8) | data[1];
310  rawData.mag[1] = ((int16_t)data[2] << 8) | data[3];
311  rawData.mag[2] = ((int16_t)data[4] << 8) | data[5];
312 
313  MAG3110_CalibrateHardIronOffset(&rawData.mag[0], &rawData.mag[1], &rawData.mag[2]);
314  /*! Read TEMP_SRC 0x0F from MAG3110. */
315  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110TempOut, &tempdata);
316  rawData.temp = tempdata;
317 
318  /* Copy Raw samples to Streaming Buffer. */
319  memcpy(streamingPacket + STREAMING_HEADER_LEN, &rawData, MAG3110_STREAM_DATA_SIZE);
320  /* Send streaming packet to Host. */
321  Host_IO_Send(streamingPacket, sizeof(streamingPacket), HOST_FORMAT_HDLC);
322  }
323 }
